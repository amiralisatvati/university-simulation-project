simulate <- function(n, pmf) {
  
  results <- numeric(n)
 
  k <- length(pmf) 
  
  for (i in 1:n) {
   
    u <- runif(1)
    

    x <- 1          
    cdf <- pmf[x] 
    
  while (u > cdf && x < k) {
      x <- x + 1
      cdf <- cdf+ pmf[x]
    }
    
    results[i] <- x
  }
  
  return(results)
}

simulate_hist <- function(n, h, F_h) {

  U <- runif(n)
  

  j <- findInterval(U, F_h, all.inside = TRUE)
  
    h_left <- h[j]  
  h_right <- h[j+1]   
  F_left <- F_h[j]  
  F_right <- F_h[j+1]   

  
  slope_h <- h_right - h_left  
  slope_F <- F_right - F_left  
  

  ratio <- (U - F_left) / slope_F
  

  ratio[slope_F == 0] <- 0
  

  X <- h_left + ratio * slope_h
  
  return(X)
}
